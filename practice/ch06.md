  answer:  
  ```cpp
  #include <iostream>
#include <string>

using std::string;
using std::cout;
using std::endl;

bool isHasUpper(const string &str){
    for(auto c:str){
        if(c >= 'A' && c <= 'Z') return true;
    }
    return false;
}

void toLower(string &str){
    for(auto &c:str)
        if(c >= 'A'  && c <= 'Z')
            c = 'a' + (c - 'A');
}

int main(){
    string str = "AbCdEf";
    cout<<isHasUpper(str)<<endl;
    toLower(str);
    cout<<str;
    return 0;
}
```
  不相同，因为一个不需要改变字符串的值，一个需要改变。
