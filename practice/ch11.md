### 练习11.3
  question:编写你自己的单词计数程序   
  answer:  
  ```cpp
  #include <iostream>
#include <map>
#include <string>

using std::string;
using std::cin;
using std::cout;
using std::endl;
using std::map;

int main(){
    map<string,int> m;
    string str;
    while(cin>>str){
        m[str]++;
    }
    for(auto pa:m)
        cout<<pa.first<<" "<<pa.second<<endl;
    return 0;
}
```
