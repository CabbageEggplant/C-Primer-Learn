### 练习11.3
  question:编写你自己的单词计数程序   
  answer:  
  ```cpp
  #include <iostream>
#include <map>
#include <string>

using std::string;
using std::cin;
using std::cout;
using std::endl;
using std::map;

int main(){
    map<string,int> m;
    string str;
    while(cin>>str){
        m[str]++;
    }
    for(auto pa:m)
        cout<<pa.first<<" "<<pa.second<<endl;
    return 0;
}
```

### 练习11.12
  question:编写程序，读入string和int序列，将每个string和int存入一个pair中，pair保存在一个vector中   
  answer:   
  ```cpp
  #include <iostream>
#include <vector>
#include <string>

using std::cin;
using std::cout;
using std::endl;
using std::string;
using std::vector;
using std::pair;

int main(){
    vector<pair<string,int>>v;
    string str;
    int n;
    while(cin>>str>>n){
        v.push_back(make_pair(str,n));
    }
    for(auto p:v)
        cout<<p.first<<" "<<p.second<<endl;
    return 0;
}
```
