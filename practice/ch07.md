### 练习7.27
  question:给你自己的Screen类添加move、set和sidplay函数，通过执行下面的代码检验你的类是否正确。  
  ```cpp
  Screen myScreen(5,5,'X');
  myScreen.move(4,0).set('#').display(cout);
  cout<<"\n";
  myScreen.display(cout);
  cout<<"\n";
  ```
  answer:  
  Screen.h
  ```cpp
  
#ifndef Screen_H

#define Screen_H

#include <string>
#include <iostream>
using std::string;


class Screen{
public:
    typedef std::string::size_type pos;
private:
    pos cursor = 0;
    pos height = 0, width = 0;
    std::string contents;
public:
    Screen() = default;
    Screen(pos height, pos width);
    Screen(pos height, pos width, char ch);
    Screen &move(pos r, pos c);
    Screen &set(char);
    Screen &set(pos, pos, char);
    const Screen &do_display(std::ostream &os) const;
    Screen &display(std::ostream &os);
    const Screen &display(std::ostream &os) const;
};

#endif
```
  Screen.cpp
  ```cpp
  #include "Screen.h"

Screen::Screen(pos height, pos width):height(height), width(width), contents(width*height,' '){}
Screen::Screen(pos height, pos width, char ch):height(height), width(width), contents(height*width,ch){}

Screen& Screen::move(pos r, pos c){
    pos row = r * width;
    cursor = row + c;
    return *this;
}

Screen& Screen::set(char c){
    contents[cursor] = c;
    return *this;
}

Screen& Screen::set(pos r, pos col, char ch){
    contents[r*width + col] = ch;
    return *this;
}

const Screen& Screen::do_display(std::ostream &os)const{
    for(int i = 0; i < height; ++i){
        for(int j = 0; j < width; ++j){
            std::cout<<contents[i*width+j];
        }
        std::cout<<std::endl;
    }
    return *this;
}

Screen& Screen::display(std::ostream &os){
    do_display(os);
    return *this;
}

const Screen& Screen::display(std::ostream &os) const{
    do_display(os);
    return *this;
}
```
  7_27.cpp
  ```cpp
  #include <iostream>
#include "Screen.h"

int main(){
    Screen myScreen(5,5,'X');
    myScreen.move(4,0).set('#').display(cout);

    cout<<"\n";
    return 0;
}
```
